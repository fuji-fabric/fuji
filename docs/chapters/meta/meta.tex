\LevelZero{Meta}\label{ch:meta}
\LevelOne{Definition}
A \ttt{meta} is a \ttt{key-value pair} introduced by \ttt{luckperms} mod, to store \ttt{per-player} or \ttt{per-group} variables.

\begin{tips}{List all metas.}
    Issue \cmd{/fuji inspect permissions-and-metas} command.
\end{tips}

\LevelOne{Example}
\begin{example}{Set a meta for a gruop.}
    The \str{home} module provides the meta \str{fuji.home.home\_limit}, which controls how many homes a player can create. \\
    To set the homes limit to \ttt{3}: \cmd{/lp group default meta set fuji.home.home\_limit 3}
\end{example}

\begin{example}{List all metas of a group.}
    Issue \cmd{/lp group default info} command.
\end{example}
