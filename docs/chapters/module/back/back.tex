\LevelOne{back}
\LevelTwo{Purpose}
This module allows player to teleport back to \tbf{last teleport point} or \tbf{death point}.

\LevelTwo{Command}
\fcmd{/back}

\LevelTwo{Configuration}
\begin{Configuration}
    \item[ignore\_distance]{
        If the player's teleportation destination is close enough, we ignore this teleportation.
    }
\end{Configuration}

\LevelTwo{Meta}
\begin{description}
    \item [fuji.back.max\_location\_entries\_to\_save] The max back location entries to save.
\end{description}
