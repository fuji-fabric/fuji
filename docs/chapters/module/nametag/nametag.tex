\LevelOne{nametag}
\LevelTwo{Purpose}
This module provides nametag customization.

\LevelTwo{Example}
\begin{example}{Set background to blue color}
    \begin{jsoncode}
        "background": -16776961
    \end{jsoncode}
\end{example}

\begin{example}{Set half transparency}
    \begin{jsoncode}
        "text_opacity": 128
    \end{jsoncode}
\end{example}

\begin{example}{Scale the size of text into double}
    \begin{jsoncode}
        "scale": {
            "x": 2.0,
            "y": 2.0,
            "z": 2.0
        },
    \end{jsoncode}

\end{example}
