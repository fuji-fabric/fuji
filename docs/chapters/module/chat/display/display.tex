\LevelThree{display}
\LevelFour{Purpose}
This module allows players to share their main hand item, their inventory and their enderchest.

\LevelFour{Purpose}
\fcmd{/chat display item}
\fcmd{/chat display inv}
\fcmd{/chat display ender}

\LevelFour{Placeholder}
\begin{description}
    \item [\ph{fuji:item}] display player's main-hand item.
    \item[\ph{fuji:inv}] display player's inventory.
    \item[\ph{fuji:ender}] display player's ender-chest.
\end{description}

\LevelFour{Configuration}
\begin{Configuration}
    \item[expiration\_duration\_s]{
        For each display data, how long should we save in the memory.
        Note that if a player shares its inventory items, then fuji will save a copy of his inventory data in the memory.
    }

\end{Configuration}

\LevelFour{Example}
\begin{example}{Define a chat shortcut to create display.}
    You can use~\nameref{ch:chat_replace} module, to define chat shortcut like \str{item}, \str{inv} and \str{ender}, and replace them with display placeholders.
    In this case, you can type \str{item}, \str{inv} and \str{ender} in chat message direct, to create a display, without issuing the command.
\end{example}
