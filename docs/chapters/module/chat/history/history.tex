\LevelThree{history}
\LevelFour{Purpose}
Store \ttt{chat messages} as a history, and send them to a player joined the server.

\begin{tips}{Works with Styled Chat mod}
    The \ttt{chat.history} module works with \ttt{chat.style} module.
\end{tips}

\LevelFour{Configuration}
\begin{Configuration}
    \item[buffer\_size]{
        How many chat messages should we save, so that we can send for a new-joined player.
    }
\end{Configuration}
