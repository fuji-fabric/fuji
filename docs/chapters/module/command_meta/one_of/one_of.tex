\LevelThree{one\_of}\label{ch:one_of}

\LevelFour{Purpose}
The one-of command will pick a random command from commands, and execute it as console.

\LevelFour{Command}
\fcmd{/one-of}

\LevelFour{Example}
\begin{example}{Pick a lucky command.}
    \cmd{/one-of say 1 one-of say 2 one-of say lucky \ph{player:name}}
\end{example}
