\LevelOne{Configuration File}

\LevelTwo{Definition}
All \tbf{files} inside \configdir directory are named \tbf{configuration files}.

\begin{tips}{List all configuration files.}
    Issue \cmd{/fuji inspect configurations} command.
\end{tips}

\LevelTwo{Types}
\begin{note}{There are two types of configuration files.}
    \begin{description}
        \item[Control File] {A \ttt{control file} is used to control behaviours.}
        \begin{description}
            \item[Main-Control File] The \tbf{main-control file} is the \mainconfig file, which is used to enable/disable a \tbf{module}.
            \item[Module-Control File] A module can have its own control file, which is used to control the behaviour of that module.
        \end{description}

        \item[User-Data File] A module can have its user-data files, they are used to store the data generated by users.
    \end{description}
\end{note}

\LevelTwo{Example}

\begin{example}{The main control file.}
    The main control file is the \mainconfig file. \\
    It's applied to \tbf{all} modules.
\end{example}

\begin{example}{The module control file.}
    The module control file of \str{afk} module is the \tbf{config/fuji/modules/afk/config.json} file. \\
    It's applied to \str{afk} module.
\end{example}

\LevelOne{Module}

\LevelTwo{Definition}
A \ttt{module} is a self-contained and disable-able unit, to provide a specific function.

\begin{example}{The function of modules}
    The \str{chat} module, provides chat-related functions. \\
    The \str{tpa} module, provides \cmd{/tpa} command.
\end{example}

\begin{tips}{List all modules.}
    Issue \cmd{/fuji inspect modules} command.
\end{tips}


\LevelTwo{Module Path}
A \tbf{module} can be identified by a unique \tbf{module path}.

\begin{example}{What does a module path look like?}
    For example:
    \begin{enumerate}
        \item {The module path of the module \str{tpa} is \str{tpa}.}
        \item {The module path of the module \str{history} whose parent module is \str{chat}, is \str{chat.history}.}
    \end{enumerate}
\end{example}

\LevelTwo{How to enable/disable a module?}

To enable an interesting target module, follow the steps:
\begin{enumerate}
    \item {Open the \mainconfig file.}
    \item {For the interesting module, set the value of \str{enable} to be \str{true} to enable it.}
    \item {Re-start the server to apply the changes.}
\end{enumerate}

\begin{warn}{Remember to restart the server to apply changes.}
    You can \tbf{NOT} enable or disable a module, while the server \tbf{is running}. \\
    After the value of \str{enable} is modified, you \tbf{must} restart the server to apply the changes. \\
    In other words, modules are only loaded during server startup.
\end{warn}

\begin{tips}{How to enable a sub-module?}
    To enable the \str{chat.display} module, you must enable \str{chat} module first.
\end{tips}

\begin{tips}{How to know what modules are enabled?}
    The \cmd{/fuji} command is an alias for the \cmd{/fuji inspect modules} command. \\
    Issue \cmd{/fuji} command, to see what modules you have enabled.
\end{tips}

\LevelOne{Command}\label{sec:command}

\LevelTwo{Definition}
A \ttt{command} is a \ttt{chat string} that starts with \str{/} string. \\
It's used to perform a specific action/function.\\

\begin{tips}{List all commands.}
    Issue \cmd{/fuji} command, to list all the enabled modules.\\
    Issue \cmd{/fuji inspect fuji-commands}, to list all the provided commands from enabled modules.
\end{tips}

\LevelTwo{Example}
\begin{example}{What does a command look like?}
    For example:
    \begin{enumerate}
        \item {\cmd{/give @s minecraft:apple}}
        \item {\cmd{/tp @r}}
        \item {\cmd{/fuji}}
        \item {...}
    \end{enumerate}
\end{example}

\LevelOne{Placeholder}

\LevelTwo{Definition}
A \ttt{placeholder} is a string that starts and ends with \str{\%} string. \\
It's used to hold a place, and its content will be replaced based on context.

\begin{tips}{List all placeholders.}
    Issue \cmd{/fuji inspect placeholders} command.
\end{tips}

\LevelTwo{Example}
\begin{example}{What does a placeholder look like?}
    For example:
    \begin{enumerate}
        \item {The placeholder \ph{player:name} will be replaced with the name of the contextual player.}
        \item {The placeholder \ph{world:name} will be replaced with the name of the contextual world.}
    \end{enumerate}
\end{example}

\begin{tips}{Check the available placeholders}
    The \tbf{PlaceholderAPI} mod is \tbf{embedded} in this mod. \\
    By default, it provides a collection of built-in placeholders. \\
    You can add more placeholders by installing additional extension mods.
    \begin{enumerate}
        \item \url{https://placeholders.pb4.eu/user/default-placeholders/}
        \item \url{https://placeholders.pb4.eu/user/mod-placeholders/}
    \end{enumerate}
\end{tips}

\LevelOne{Identifier}

\LevelTwo{Definition}
An \ttt{identifier} is a \ttt{string} to name an \ttt{object} in \ttt{Minecraft}.

\begin{tips}{List all identifiers.}
    Issue \cmd{/fuji inspect registry} command.
\end{tips}

\LevelTwo{Example}

\begin{example}{The identifier of an item.}
    The identifier of apple is \str{minecraft:apple}.\\
    The identifier of diamond is \str{minecraft:diamond}.
\end{example}

\begin{example}{The identifier of an entity.}
    The identifier of zombie is \str{minecraft:zombie}.
\end{example}

\LevelOne{Job}\label{sec:job}

\LevelTwo{Definition}
A \ttt{job} is some things that will be done \ttt{repeatedly}.

\begin{tips}{List all jobs.}
    Issue \cmd{/fuji inspect jobs} command.
\end{tips}

\LevelTwo{Schedule Table}
The schedule table of a job is a calendar. \\
A string named \tbf{cron expression} is used to \tbf{select} time points in the calendar. \\
When the selected time points are reached, the job will be \tbf{triggered/fired}. \\
When a job is triggered/fired, the things about that job will be done at once.

\begin{example}{What does a cron expression look like?}
    For example:
    \begin{enumerate}
        \item {The cron expression \str{0 * * ? * *} means \str{trigger the job every minute}.}
        \item {The cron expression \str{0 */3 * ? * *} means \str{trigger the job every 3 minutes}.}
    \end{enumerate}
\end{example}

\begin{tips}{Use a generator to make a cron expression.}
    You can use a generator to select the desired time points:
    \url{https://www.freeformatter.com/cron-expression-generator-quartz.html}
\end{tips}

\LevelOne{Regex}\label{term:regex}

\LevelTwo{Definition}
A \tbf{regex expression} is a \tbf{pattern string}, used to match a set of \tbf{input strings}.

\LevelTwo{Example}
\begin{example}{What does a regex expression look like?}
    For example:
    \begin{enumerate}
        \item {The pattern string \str{cat} is used to match exactly the given input string \str{cat}.}
        \item {The pattern string \str{cat|dog} is used to match the given input strings \str{cat} or \str{dog}.}
    \end{enumerate}
\end{example}

\begin{tips}{Test your pattern string on the fly.}
    Use the following online editors, to test your \tbf{pattern string}.
    \begin{enumerate}
        \item \url{https://regexr.com/}
        \item \url{https://regex101.com/}
    \end{enumerate}
\end{tips}
