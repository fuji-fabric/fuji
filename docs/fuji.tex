% Preamble
\documentclass[10pt,oneside]{book}

% Packages

% Use a4 paper size.
\usepackage[a4paper, total={6in, 8in}]{geometry}

% Use color extension package.
\usepackage{xcolor}

% Use colorbox package.
\usepackage{tcolorbox}

% Configure hyper link colors.
\usepackage{hyperref}
\hypersetup{
    colorlinks=true,
    linkcolor=blue,
    urlcolor=red
}

% The 'outputdir' is reletive to fuji.tex, and must point to the 'output directory' of latex compiler.
\usepackage{minted}

% Fix the issue that the character '_' can't be copied in output .pdf file.
\usepackage[T1]{fontenc}

% Disable double hyphen ligature in typeset.
\usepackage{microtype}
\DisableLigatures[-]{}

% Use object listing package.
\usepackage{listings}

% Define macros.
\input{macros/macro}

% Use fancy header and footer package.
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhead[LE,RO]{\nouppercase{\rightmark}}
\fancyhead[LO,RE]{\textbf{\nouppercase{\leftmark}}}
\fancyfoot[C]{\thepage}

% Document
\begin{document}

    \title{
        \textbf{Fuji Manual}\\[0.5em]
        \large {\version}
    }
    \author{fuji-fabric}
    \date{\today}
    \maketitle

    % Make the TOC.
    \tableofcontents
    \mainmatter

    % Include the chapters.
    \input{chapters/concept/concept}
    \input{chapters/permission/permission}
    \input{chapters/meta/meta}
    \input{chapters/language/language}
    \input{chapters/qna/qna}
    \input{chapters/suggestion/suggestion}

\end{document}
